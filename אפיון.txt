## 🧠 service – לוגיקה

### services/leadsService.js

```js
const supabase = require('../config/supabase')

exports.createLead = async (data) => {
  return await supabase.from('leads').insert([data])
}

exports.getLeads = async () => {
  return await supabase.from('leads').select('*').order('created_at', { ascending: false })
}

exports.updateLead = async (id, data) => {
  return await supabase.from('leads').update(data).eq('id', id)
}

exports.deleteLead = async (id) => {
  return await supabase.from('leads').delete().eq('id', id)
}
```

---

## 🎯 controller

### controllers/leadsController.js

```js
const leadsService = require('../services/leadsService')

exports.createLead = async (req, res) => {
  try {
    const { full_name, phone } = req.body

    if (!full_name || !phone) {
      return res.status(400).json({ error: 'Missing required fields' })
    }

    const result = await leadsService.createLead({
      ...req.body,
      status: 'new'
    })

    res.json({ success: true, data: result })
  } catch (err) {
    res.status(500).json({ error: 'Server error' })
  }
}

exports.getLeads = async (req, res) => {
  const result = await leadsService.getLeads()
  res.json(result)
}

exports.updateLead = async (req, res) => {
  const { id } = req.params
  const result = await leadsService.updateLead(id, req.body)
  res.json(result)
}

exports.deleteLead = async (req, res) => {
  const { id } = req.params
  const result = await leadsService.deleteLead(id)
  res.json(result)
}
```

## 🔗 routes

### routes/leads.js

```js
const express = require('express')
const router = express.Router()
const controller = require('../controllers/leadsController')

router.post('/', controller.createLead)
router.get('/', controller.getLeads)
router.put('/:id', controller.updateLead)
router.delete('/:id', controller.deleteLead)

module.exports = router
```

# 🌐 שלב 3 – חיבור טופס בדף נחיתה

## דוגמה לטופס

```html
<form id="leadForm">
  <input type="text" placeholder="שם מלא" name="full_name" required />
  <input type="tel" placeholder="טלפון" name="phone" required />
  <input type="email" placeholder="אימייל" name="email" />
  <textarea name="message"></textarea>

  <button type="submit">שלח</button>
</form>

<script>
document.getElementById('leadForm').addEventListener('submit', async (e) => {
  e.preventDefault()

  const form = e.target

  const data = {
    full_name: form.full_name.value,
    phone: form.phone.value,
    email: form.email.value,
    message: form.message.value,
    page_source: 'landing-classa'
  }

  await fetch('https://YOUR-RENDER-URL/api/leads', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json'
    },
    body: JSON.stringify(data)
  })

  alert('נשלח בהצלחה')
})
</script>
```
